#version 330

layout (location = 0) in vec3 in_position;
layout (location = 1) in vec4 in_color;
layout (location = 2) in vec2 in_uv;

out vec2 position;
out vec4 color;
out vec2 uv;

uniform mat4 transform;

void main() {
	gl_Position = transform * vec4(in_position, 1.0);
	
	position = in_position;
	color = in_color;
	uv = in_uv;
}
